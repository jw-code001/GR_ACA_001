CREATE TYPE q_type AS ENUM ('r', 'c', 't');

CREATE TABLE survey_questions (
  question_no int4 PRIMARY KEY,
  content text NOT NULL,
  question_type q_type NOT NULL, -- (r, c, t)
  
  -- 선택지 뭉치 (예: "10대, 20대, 30대")
  options_summary text, 
  
  is_active boolean DEFAULT true,
  created_at timestamp with time zone DEFAULT now(),

  -- [강력한 제약 조건]
  CONSTRAINT check_options_logic CHECK (
    -- 1. 텍스트 타입(t)은 선택지가 없어도 됨 (NULL 허용)
    (question_type = 't') 
    OR 
    (
      -- 2. 라디오(r)와 체크박스(c)인 경우
      question_type IN ('r', 'c') AND 
      options_summary IS NOT NULL AND 
      -- 공백 제거 후 내용이 있어야 함
      length(trim(options_summary)) > 0 AND
      -- ★ 핵심: 콤마(,)가 반드시 포함되어야 함 (최소 2개 이상의 선택지 강제)
      options_summary LIKE '%,%'
    )
  )
);
